Autocomplete

Autocomplete form component provides suggestions while you type. It is often used for tags and contacts input.

Selected: [ ]

<script>
    import { Autocomplete } from 'svelte-spectre'

    let predefined = [
            'top_left',
            'top_center',
            'top_right',
            'center_left',
            'center_center',
            'center_right',
            'bottom_left',
            'bottom_center',
            'bottom_right',
        ],
        groups = [
            '↖ top_left',
            '↑ top_center',
            '↗ top_right',
            '← center_left',
            '✛ center_center',
            '→ center_right',
            '↙ bottom_left',
            '↓ bottom_center',
            '↘ bottom_right',
        ]

    const groupsBy = (item) => {
        return groups.split(', ').find((g) => item.includes(g))
    }
</script>

<Autocomplete {predefined} bind:selected {groupsBy} />
<small>Selected: [ {selected.join(", ")} ]</small>
